// compiler_test.txt

func main() -> void
  a = 10;
  b = 20;
  write(a + b); // Should print 30.0
  write("\n");

  s1 = "Hello";
  s2 = " World!";
  s3 = s1 + s2;
  write(s3); // Should print "Hello World!"
  write("\n");

  s_rep = "abc" * 3;
  write(s_rep); // Should print "abcabcabc"
  write("\n");

  if a > b then
    write("a is greater");
  else
    write("b is greater"); // Should print "b is greater"
  end
  write("\n");

  c = 0;
  while c < 3 do
    write(c);
    write(" ");
    c = c + 1;
  end
  write("End while\n");

  for i from 0 to 2 do
    write(i);
    write(" ");
  end
  write("End for\n");

  l = [1.1, 2.2, 3.3];
  write(l[0]); // Should print 1.1
  write("\n");

  l << 4.4; // Append
  write(l[3]); // Should print 4.4
  write("\n");

  val = dequeue from l;
  write(val); // Should print 1.1
  write("\n");
  write(l[0]); // Should print 2.2
  write("\n");

  my_lambda = lambda(x) -> f64 x * 2;
  res_lambda = my_lambda(5);
  write(res_lambda); // Should print 10.0
  write("\n");
end

func process_data(data)
    write("Processing: ");
    write(data);
    write("\n");
    return data + 10;
end
